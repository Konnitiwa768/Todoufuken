# カオスイベント101: ランダム戦争
event = {
    id = chaos.101
    title = "ランダム戦争勃発"
    desc = "世界のどこかで戦争が始まった！"
    picture = GFX_report_event_generic

    immediate = {
        random_country = {
            limit = { is_major = yes }
            create_wargoal = {
                type = take_state_focus
                target = random_country = { NOT = { tag = PREV } }
            }
        }
    }

    option = {
        name = "戦争が始まった！"
    }
}
